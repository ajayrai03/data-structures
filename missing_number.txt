class Solution {
    public int missingNumber(int[] nums) {
        
        int sum1=0;
        int sum2=0;
        int n=nums.length;
        for(int i=1;i<=n;i++){
            sum1=sum1+i;
        }
        for(int i=0;i<nums.length;i++){
            sum2=sum2+nums[i];
        }
        int result=sum1-sum2;
        return result;
    }
}
